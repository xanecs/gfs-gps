def shift(register, feedback, output):

    out = [register[i-1] for i in output]
    if len(out) > 1:
        out = sum(out) % 2
    else:
        out = out[0]

    fb = sum([register[i-1] for i in feedback]) % 2

    for i in reversed(range(len(register[1:]))):
        register[i+1] = register[i]

    register[0] = fb

    return out
